```python
def solution(skill, skill_trees):
    answer = 0
    for sk_tr in skill_trees:
        order = ''
        for sk in sk_tr:
            if sk not in skill:
                continue
            order += sk
        if order == skill[:len(order)]:
            answer += 1
    return answer
```
