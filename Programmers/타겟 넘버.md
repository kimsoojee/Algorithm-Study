```python
def solution(numbers, target):
    q = [numbers[0],numbers[0]*-1]
    for n in numbers[1:]:
        temp = []
        for x in q:
            temp.append(x+n)
            temp.append(x-n)
        q = temp
    answer = sum([x==target for x in q])
    return answer
```
