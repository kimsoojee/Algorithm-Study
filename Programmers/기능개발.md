```python
def solution(progresses, speeds):
    answer = []
    while progresses:
        for x in range(len(progresses)):
            progresses[x] += speeds[x]
        release = 0
        while progresses and progresses[0] >= 100:
            release += 1
            progresses, speeds = progresses[1:], speeds[1:]
        if release > 0:
            answer.append(release)
    return answer
```

```python
def solution(progresses, speeds):

            Q[-1][1]+=1
```
