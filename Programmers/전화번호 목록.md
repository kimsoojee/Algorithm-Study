```python
def solution(phone_book):
    answer = True
    phone_book = sorted(phone_book)
    Hash = []
    for x in phone_book:
        if not Hash or Hash[0][0] != x[0]:
            Hash = [x]
        else:
            prefixL = min(len(x), len(Hash[0]))
            if x[:prefixL] == Hash[0][:prefixL]:
                return False
            Hash = [x]
    return answer
```
