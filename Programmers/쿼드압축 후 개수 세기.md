def partition(arr, r, c, size):
    if size == 1:
        count[arr[r][c]] += 1
        return
    value = arr[r][c]
    check = True
    for x in range(r, r+size):
        if check:
            for y in range(c, c+size):
                if arr[x][y] != value:
                    check = False
                    break
        else: break
    if check:
        count[value] += 1
    else:
        size //= 2
        partition(arr, r, c, size)
        partition(arr, r, c+size, size)
        partition(arr, r+size, c, size)
        partition(arr, r+size, c+size, size)

def solution(arr):
    answer = [0, 0]
    def check(size, x, y):
        if size == 1:
            answer[arr[y][x]] += 1
            return
        else:
            first = arr[y][x]
            for dy in range(size):
                for dx in range(size):
                    if first != arr[y + dy][x + dx]:
                        check(size // 2, x, y)
                        check(size // 2, x + size // 2, y)
                        check(size // 2, x, y + size // 2)
                        check(size // 2, x + size // 2, y + size // 2)
                        return
            answer[first] += 1
    check(len(arr),0,0)
    return answer
