```python
def solution(number, k):
    answer = []
    for i,x in enumerate(number):
        while answer and int(answer[-1]) < int(x) and k>0:
                answer.pop()
                k -= 1
        if k == 0:
            answer.append(number[i:])
            break
        answer.append(x)
    return ''.join(answer)[:len(number)-k]
```
